draw_VNC <- function(data, distance_measure){ # options: "cv", "sd"
  par(new=TRUE, cex.main=1.5, cex.lab=1.5, cex.axis=0.995)
  distance.measure <- distance_measure
  
  input <- data[, 3:ncol(data)] %>% t # 
  rownames(input) <- NULL # 
  years <- data[, 2]
  colnames(input) <- years # 
  xlabels <- years
  
  data.collector <- list()
  data.collector[["0"]] <- input
  position.collector <- list()
  position.collector[[1]] <- 0
  overall.distance <- 0
  number.of.steps <- ncol(input) - 1 # 
  
  dist_lst <- c() #
  for (i in 1:number.of.steps) {
      cat(i/number.of.steps, "\n", sep = "")
      difference.checker <- numeric()
      for (j in 1:(length(unique(colnames(input))) - 1)) {
          first.name <- unique(colnames(input))[j] # 
          second.name <- unique(colnames(input))[(j + 1)] # 
          pooled.sample.first <- input[,colnames(input) == first.name] # 
          pooled.sample.second <- input[,colnames(input) == second.name] # 
          if(distance.measure=="cor"){
            cor.dist <- 1 - cor(pooled.sample.first, pooled.sample.second) # 
            difference.checker[j] <- cor.dist # 
          }
      }
      
      pos.to.be.merged <- which.min(difference.checker)
      distance <- min(difference.checker)
      dist_lst <- c(dist_lst, distance) #
      overall.distance <- overall.distance + distance
      lower.name <- unique(colnames(input))[pos.to.be.merged] # 
      higher.name <- unique(colnames(input))[(pos.to.be.merged + # 
          1)]
      matches <- colnames(input) %in% c(lower.name, higher.name) # 
      new.mean.age <- round(mean(as.numeric(colnames(input)[colnames(input) %in%  # 
          c(lower.name, higher.name)])), 4)
      position.collector[[(i + 1)]] <- which(colnames(input) == # 
          lower.name | colnames(input) == higher.name) # 
      colnames(input)[colnames(input) %in% c(lower.name, higher.name)] <- as.character(new.mean.age) # 
      data.collector[[(i + 1)]] <- input
      names(data.collector)[(i + 1)] <- distance # 
  }

  # prepare canvas
  if (distance.measure == "cor"){
  plot(0, mean(years), xlim = range(years), ylim = c(0, 1.1 * 
    sum(as.numeric(names(data.collector)))), main = "", 
    xlab = "Year", ylab = "Correlation distance", 
    type = "n", axes = FALSE)
    
  }
  
  grid()
  axis(1, at = years, labels=xlabels)
  axis(2)
  
  new.data.collector <- lapply(data.collector, function(x){colnames(x)})
  names(new.data.collector) <- names(data.collector)
  # plotting data points
  cur.y <- rep(0, ncol(data.collector[[1]]))
  for (k in 1:(ncol(data.collector[[1]]) - 1)) {
      cur.x <- as.numeric(unique(new.data.collector[k][[1]]))
      cur.y.plus <- as.numeric(unique(names(new.data.collector[(k + 1)])))
      arrows(cur.x, cur.y, cur.x, cur.y + cur.y.plus, length = 0)
      cur.y <- rep(unique(cur.y) + cur.y.plus, length(cur.x) - 1)
      left <- min(position.collector[[(k + 1)]])
      right <- max(position.collector[[(k + 1)]])
      lower.x <- as.numeric(new.data.collector[[k]][left])
      higher.x <- as.numeric(new.data.collector[[k]][right])
      arrows(lower.x, unique(cur.y), higher.x, unique(cur.y), 
          length = 0)
  }
  print("ok!!!")
  return(rev(dist_lst))
}
create_data <- function(collo_position){
  data <- read.csv(input_fp_lst[paste0("[step 9] VNC/", collo_position, "_collocate_df.csv")]) %>%
    drop_na()
  
  data_collo <- data$collo
  data <- data %>%
    select(-collo) %>%
    t %>%
    as.data.frame() %>%
    rowid_to_column()
  colnames(data) <- c("dynasty", data_collo)
  
  data <- data %>%
    mutate(
      ID=1:nrow(data),
      YEAR=c(618, 960, 1271, 1368, 1644, 1980)
    ) %>%
    select(ID, YEAR, everything(), -dynasty)
  
  return(data)
}

    collocate_df <- create_data(position)
    
    draw_VNC(collocate_df, distance_measure="cor")
  
  dev.off
})
